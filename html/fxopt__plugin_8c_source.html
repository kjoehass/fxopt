598  fprintf(stderr, "RHS TREE CLASS: %s CODE: %s\n",
599  TREE_CODE_CLASS_STRING(xclass),
600  tree_code_name[rhs_code]);
601  }
603  // Done analyzing an assignment statement
605  if (format_initialized(result_frmt)) {
606  //
607  // Now construct and insert any necessary shift operations. This
608  // section is executed after each GIMPLE arithmetic operator is
609  // analyzed.
610  //
611  // Build a shift statement for each RHS operand. A negative shift
612  // is left, a positive shift is right. The right shift operator is
613  // a true arithmetic shift.
614  //
615  int i;
616  for (i = 1; i < num_operands; i++) {
617  if (!lastpass)
618  break;
619  tree current_oprnd_tree = oprnd_tree[i];
620  char var_name[32];
621  if (oprnd_frmt[i].shift != 0) {
622  check_shift(oprnd_frmt[i]);
623  if (!oprnd_frmt[1].ptr_op) {
624  if (oprnd_frmt[i].shift > 0)
625  fprintf(stderr, " OP%d >>%-2d", i, oprnd_frmt[i].shift);
626  else
627  fprintf(stderr, " OP%d <<%-2d", i, -oprnd_frmt[i].shift);
628  fprintf(stderr, " (%2d/%2d/%2d/%2d)", oprnd_frmt[i].S,
629  oprnd_frmt[i].I, oprnd_frmt[i].F, oprnd_frmt[i].E);
630  print_min_max(oprnd_frmt[i]);
631  }
632 
633  if (TREE_CONSTANT(oprnd_tree[i])) {
634  //
635  // If this operand is a constant, just change its value by
636  // shifting and rounding at compile time, then replace the
637  // old value with the new value
638  //
639  double_int constant_double =
640  tree_to_double_int(oprnd_tree[i]);
641 
642  if (oprnd_frmt[i].shift < 0) { // shift left
643  oprnd_frmt[i].shift = -oprnd_frmt[i].shift;
644  constant_double = double_int_lshift(constant_double,
645  (HOST_WIDE_INT)
646  oprnd_frmt[i].shift,
647  oprnd_frmt[i].size,
648  false);
649  } else { // shift right
650  double_int round_const = double_int_lshift(double_int_one,
651  (HOST_WIDE_INT)
652  (oprnd_frmt
653  [i].shift -
654  1),
655  oprnd_frmt
656  [i].size,
657  false);
658  if (double_int_negative_p(constant_double))
659  round_const =
660  double_int_add(round_const, double_int_minus_one);
661 
662  constant_double =
663  double_int_add(constant_double, round_const);
664  constant_double =
665  double_int_rshift(constant_double,
666  (HOST_WIDE_INT) oprnd_frmt[i].shift,
667  oprnd_frmt[i].size, true);
668  }
669  current_oprnd_tree =
670  double_int_to_tree(TREE_TYPE(oprnd_tree[i]),
671  constant_double);
672  } else if (MEM_REF == TREE_CODE(oprnd_tree[i])) {
673  //
674  // MEM_REF operands are actually expressions and they have
675  // two operands. The first is a pointer and the second is
676  // an offset. The offset is an integer number of bytes,
677  // and is a multiple of the size of the data pointed to
678  // by the pointer.
679  // If a real type is replaced by an integer type it may be
680  // necessary to adjust the offset for the new size of the
681  // data. The amount of the adjustment is determined in
682  // pointer_math() and passed as the shift element in the
683  // SIF format for the MEM_REF
684  //
685  double_int constant_double = mem_ref_offset(oprnd_tree[i]);
686  if (oprnd_frmt[i].shift < 0) { // shift left
687  oprnd_frmt[i].shift = -oprnd_frmt[i].shift;
688  constant_double =
689  double_int_lshift(constant_double,
690  (HOST_WIDE_INT) oprnd_frmt[i].shift,
691  oprnd_frmt[i].size, false);
692  } else { // shift right
693  constant_double =
694  double_int_rshift(constant_double,
695  (HOST_WIDE_INT) oprnd_frmt[i].shift,
696  oprnd_frmt[i].size, true);
697  }
698  tree const_tree =
699  double_int_to_tree(TREE_TYPE
700  (TREE_OPERAND(oprnd_tree[i], 1)),
701  constant_double);
702  current_oprnd_tree = build2(MEM_REF,
703  TREE_TYPE(oprnd_tree[i]),
704  TREE_OPERAND(oprnd_tree[i], 0),
705  const_tree);
706  } else {
707  //
708  // Add explicit statements to perform the shift, possibly
709  // with rounding
710  //
711  // var_to_shift is the variable that will actually, finally,
712  // be shifted, after any rounding is done
713  //
714  tree var_to_shift = oprnd_tree[i];
715  // shift_expr will be right for shift>0, left for shift<0
716  enum tree_code shift_expr;
717  if (oprnd_frmt[i].shift < 0) {
718  oprnd_frmt[i].shift = -oprnd_frmt[i].shift;
719  shift_expr = LSHIFT_EXPR;
720  } else {
721  shift_expr = RSHIFT_EXPR;
722  //
723  // Only round if the shift will cause F bits to be lost
724  //
725  if (ROUNDING
726  && (oprnd_frmt[i].originalF > oprnd_frmt[i].F)) {
727  //
728  // Check to see if rounding this operand could cause an
729  // overflow. If so, and if guarding is enabled, then do
730  // the guarding.
731  //
732  if (GUARDING && rounding_may_overflow(oprnd_frmt[i])) {
733  //
734  // When guarding, shift the operand right one place
735  // before rounding. If overflow occurs when rounding
736  // then we have 1 bit of headroom, although the result
737  // format doesn't take this into account...expect to
738  // be lucky that overflow doesn't occur with _both_
739  // operands and corrupt the result
740  //
741  sprintf(var_name, "_fx_guard%d", i);
742  tree guarded_var =
743  make_rename_temp(long_long_integer_type_node,
744  var_name);
745  new_stmt =
746  gimple_build_assign_with_ops(RSHIFT_EXPR,
747  guarded_var,
748  var_to_shift,
749  build_one_cst
750  (integer_type_node));
751  print_gimple_stmt(stderr, new_stmt, 2, 0);
752  gsi_insert_before(&gsi, new_stmt, GSI_SAME_STMT);
753  var_to_shift = guarded_var;
754  oprnd_frmt[i].S++;
755  if (oprnd_frmt[i].E > 0)
756  oprnd_frmt[i].E--;
757  else
758  oprnd_frmt[i].F--;
759  oprnd_frmt[i].shift--;
760  current_oprnd_tree = guarded_var;
761  }
762 
763  if (oprnd_frmt[i].shift > 0) {
764  // shift_constant is (1 << n-1) for n-bit shift
765  sprintf(var_name, "_fx_round%d", i);
766  tree rounded_var =
767  make_rename_temp(TREE_TYPE(oprnd_tree[i]),
768  var_name);
769  tree rounding_const =
770  build_int_cst(integer_type_node,
771  (1UL << (oprnd_frmt[i].shift - 1)));
772  //
773  // do a signed round if both positive and negative
774  // values are possible
775  // *** this requires a run-time evaluation of sign
776  //
777  if (double_int_negative_p(oprnd_frmt[i].min) &&
778  double_int_positive_p(oprnd_frmt[i].max) &&
779  !POSITIVE) {
780  // sign_bit_var is -1 if op is negative, else 0
781  sprintf(var_name, "_fx_signbit%d", i);
782  tree sign_bit_var =
783  make_rename_temp(TREE_TYPE(oprnd_tree[i]),
784  var_name);
785  tree sign_shift_cst =
786  build_int_cst(integer_type_node,
787  (oprnd_frmt[i].size - 1));
788  new_stmt =
789  gimple_build_assign_with_ops(RSHIFT_EXPR,
790  sign_bit_var,
791  var_to_shift,
792  sign_shift_cst);
793  print_gimple_stmt(stderr, new_stmt, 2, 0);
794  gsi_insert_before(&gsi, new_stmt, GSI_SAME_STMT);
795  // Now we add the rounding_const and sign_bit_var
796  sprintf(var_name, "_fx_sround%d", i);
797  tree s_round_var =
798  make_rename_temp(TREE_TYPE(oprnd_tree[i]),
799  var_name);
800  new_stmt =
801  gimple_build_assign_with_ops(PLUS_EXPR,
802  s_round_var,
803  sign_bit_var,
804  rounding_const);
805  print_gimple_stmt(stderr, new_stmt, 2, 0);
806  //gimple_set_visited(new_stmt, true);
807  gsi_insert_before(&gsi, new_stmt, GSI_SAME_STMT);
808  // Add the adjusted round value to op
809  new_stmt =
810  gimple_build_assign_with_ops(PLUS_EXPR,
811  rounded_var,
812  var_to_shift,
813  s_round_var);
814  //
815  // Only positive values are possible
816  //
817  } else if (double_int_positive_p(oprnd_frmt[i].max)) {
818  new_stmt =
819  gimple_build_assign_with_ops(PLUS_EXPR,
820  rounded_var,
821  var_to_shift,
822  rounding_const);
823  //
824  // Only negative values are possible
825  //
826  } else {
827  rounding_const =
828  build_int_cst(integer_type_node,
829  (1UL <<
830  (oprnd_frmt[i].shift - 1)) - 1);
831  new_stmt =
832  gimple_build_assign_with_ops(PLUS_EXPR,
833  rounded_var,
834  var_to_shift,
835  rounding_const);
836  }
837  print_gimple_stmt(stderr, new_stmt, 2, 0);
838  gsi_insert_before(&gsi, new_stmt, GSI_SAME_STMT);
839  var_to_shift = rounded_var;
840  } // shift > 0 after guarding
841  } // if rounding
842  } // if right shift
843  if (oprnd_frmt[i].shift != 0) {
844  sprintf(var_name, "_fx_shft%d", i);
845  tree shift_var =
846  make_rename_temp(TREE_TYPE(oprnd_tree[i]), var_name);
847  tree shift_constant = build_int_cst(integer_type_node,
848  oprnd_frmt[i].shift);
849  new_stmt =
850  gimple_build_assign_with_ops(shift_expr, shift_var,
851  var_to_shift,
852  shift_constant);
853  print_gimple_stmt(stderr, new_stmt, 2, 0);
854  gsi_insert_before(&gsi, new_stmt, GSI_SAME_STMT);
855  current_oprnd_tree = shift_var;
856  }
857  }
858  }
859  //
860  // If the current operand was modified, modify the statement
861  //
862  if (current_oprnd_tree != oprnd_tree[i]) {
863  gimple_set_op(stmt, i, current_oprnd_tree);
864  gimple_set_modified(stmt, true);
865  }
866  }
867  //
868  // //////// Assign a format to the result variable //////////
869  //
870  // If the desired result format is larger than the actual oprnd0
871  // variable, create a new variable for the desired result and
872  // update the original statement accordingly.
873  //
874  tree result_var = oprnd_tree[0];
875  if (result_frmt.size > oprnd_frmt[0].size) {
876  //
877  // Replace the original LHS with a wider variable
878  //
879  tree wider_var; // need this name in scope later
880  if (lastpass) {
881  wider_var =
882  make_rename_temp(long_long_integer_type_node,
883  "_fx_wide0");
884  gimple_set_op(stmt, 0, wider_var);
885  gimple_set_modified(stmt, true);
886  update_stmt(stmt);
887  print_gimple_stmt(stderr, stmt, 2, 0);
888  result_var = wider_var;
889  }
890 
891  if (ROUNDING && GUARDING && (result_frmt.S == 1)) {
892  if (lastpass) {
893  tree guarded_var =
894  make_rename_temp(long_long_integer_type_node,
895  "_fx_guard0");
896  new_stmt =
897  gimple_build_assign_with_ops(RSHIFT_EXPR, guarded_var,
898  result_var,
899  build_one_cst
900  (integer_type_node));
901  print_gimple_stmt(stderr, new_stmt, 2, 0);
902  gsi_insert_after(&gsi, new_stmt, GSI_NEW_STMT);
903  result_var = guarded_var;
904  }
905  result_frmt.S++;
906  if (result_frmt.E > 0)
907  result_frmt.E--;
908  else
909  result_frmt.F--;
910  }
911  //
912  // Explicitly cast the new wider LHS to the width of the
913  // original LHS. Shift right so there remains just 1 sign bit in
914  // the lower half of the double-precision result
915  //
916  result_frmt.shift = oprnd_frmt[0].size - result_frmt.S + 1;
917 
918  if (result_frmt.shift > 0) {
919  // don't round if there are enough empty bits
920  if (ROUNDING && lastpass) {
921  tree rounded_var =
922  make_rename_temp(TREE_TYPE(wider_var), "_fx_round0");
923  // shift_constant is (1 << n-1) for n-bit shift
924  // unless guarding, then it is (1 << n-2) FIXME?
925  // if saturating to fxfrmt, adjust shift
926  int half_lsb_shift = result_frmt.shift - 1;
927  if (format_initialized(oprnd_frmt[0]) &&
928  (oprnd_frmt[0].I < result_frmt.I)) {
929  half_lsb_shift -= result_frmt.I - oprnd_frmt[0].I;
930  half_lsb_shift = MAX(half_lsb_shift, 0);
931  }
932  tree rounding_const;
933  if (result_frmt.shift > HOST_BITS_PER_WIDE_INT) {
934  half_lsb_shift -= HOST_BITS_PER_WIDE_INT;
935  rounding_const =
936  build_int_cst_wide(integer_type_node, 0,
937  (1UL << half_lsb_shift));
938  } else {
939  rounding_const =
940  build_int_cst_wide(integer_type_node,
941  (1UL << half_lsb_shift), 0);
942  }
943  if (double_int_negative_p(result_frmt.min) && !POSITIVE) {
944  // round signed
945  // sign_bit_var is -1 if op is negative, else 0
946  // this must be a run-time check
947  tree sign_bit_var = make_rename_temp(TREE_TYPE(wider_var),
948  "_fx_signbit0");
949  tree sign_shifting_const =
950  build_int_cst(integer_type_node,
951  result_frmt.size - 1);
952  new_stmt =
953  gimple_build_assign_with_ops(RSHIFT_EXPR,
954  sign_bit_var,
955  result_var,
956  sign_shifting_const);
957  print_gimple_stmt(stderr, new_stmt, 2, 0);
958  gsi_insert_after(&gsi, new_stmt, GSI_NEW_STMT);
959  // Now we add the rounding_const and the sign_bit_var
960  tree s_round_var = make_rename_temp(TREE_TYPE(wider_var),
961  "_fx_sround0");
962  new_stmt =
963  gimple_build_assign_with_ops(PLUS_EXPR,
964  s_round_var,
965  sign_bit_var,
966  rounding_const);
967  gsi_insert_after(&gsi, new_stmt, GSI_NEW_STMT);
968  print_gimple_stmt(stderr, new_stmt, 2, 0);
969  // Add the adjusted round value to op
970  new_stmt =
971  gimple_build_assign_with_ops(PLUS_EXPR, rounded_var,
972  result_var, s_round_var);
973  } else {
974  // round unsigned
975  new_stmt =
976  gimple_build_assign_with_ops(PLUS_EXPR, rounded_var,
977  result_var,
978  rounding_const);
979  }
980  gsi_insert_after(&gsi, new_stmt, GSI_NEW_STMT);
981  print_gimple_stmt(stderr, new_stmt, 2, 0);
982  result_var = rounded_var;
983  } // end of rounding
984  //
985  // Saturate if necessary, after rounding
986  // Necessary means that the orignal LHS has an fxfrmt
987  // attribute AND either (the number of I bits in the result
988  // is different) OR (the LHS is a pointer and the number of
989  // sign bits is different)
990  // For non-pointers we only saturate the magnitude but for
991  // pointers also force the binary point to the same location
992  //
993  result_frmt =
994  apply_fxfrmt(&gsi, oprnd_frmt, oprnd_tree, result_frmt,
995  &result_var);
996  } // end of shift to cast wide to narrow
997  //
998  // Saturate if necessary...FIXME is it ever necessary?
999  //
1000  if (oprnd_frmt[0].has_attribute
1001  && ((oprnd_frmt[0].I != result_frmt.I)
1002  || (oprnd_frmt[0].ptr_op
1003  && (result_frmt.S != oprnd_frmt[0].S)))) {
1004  fprintf(stderr, " !! Second call to apply_fxfrmt\n");
1005  result_frmt =
1006  apply_fxfrmt(&gsi, oprnd_frmt, oprnd_tree,
1007  result_frmt, &result_var);
1008  }
1009  //
1010  // Do the actual cast to the original, smaller size
1011  //
1012  //FIXME fails if original stmt was not a NOP??
1013  if (lastpass) {
1014  new_stmt =
1015  gimple_build_assign_with_ops(CONVERT_EXPR,
1016  oprnd0_tree, result_var,
1017  NULL);
1018  gsi_insert_after(&gsi, new_stmt, GSI_NEW_STMT);
1019  print_gimple_stmt(stderr, new_stmt, 2, 0);
1020  }
1021  //
1022  // The result now fits in the LHS operand
1023  // Update the format based on the actual shift that was done
1024  //
1025  result_frmt = new_range(result_frmt);
1026  result_frmt.size = oprnd_frmt[0].size;
1027  result_frmt.S = 1;
1028  if (result_frmt.shift > result_frmt.E) {
1029  result_frmt.F =
1030  result_frmt.F + result_frmt.E - result_frmt.shift;
1031  result_frmt.E = 0;
1032  } else {
1033  result_frmt.E -= result_frmt.shift;
1034  }
1035  result_frmt.shift = 0;
1036  } else {
1037  //
1038  // Original operand0 and result are same size
1039  // If necessary, saturate to the desired oprnd0 format
1040  //
1041  //FIXME if original stmt was unary, should this stmt be NOP
1042  // instead of, say an array reference or SSA_NAME?
1043  if (oprnd_frmt[0].has_attribute
1044  && ((oprnd_frmt[0].I != result_frmt.I)
1045  || (oprnd_frmt[0].S != result_frmt.S))) {
1046  if (lastpass) {
1047  tree unsat_var =
1048  make_rename_temp(TREE_TYPE(oprnd0_tree), "_fx_unsat0");
1049  gimple_set_op(stmt, 0, unsat_var);
1050  gimple_set_modified(stmt, true);
1051  update_stmt(stmt);
1052  print_gimple_stmt(stderr, stmt, 2, 0);
1053  result_var = unsat_var;
1054  }
1055 
1056  result_frmt =
1057  apply_fxfrmt(&gsi, oprnd_frmt, oprnd_tree, result_frmt,
1058  &result_var);
1059 
1060  if (lastpass && (oprnd0_tree != result_var)) {
1061  if ((rhs_code == SSA_NAME) || (rhs_code == CONVERT_EXPR) ||
1062  (rhs_code == ARRAY_REF)) {
1063  new_stmt =
1064  gimple_build_assign_with_ops(rhs_code, oprnd0_tree,
1065  result_var, NULL);
1066  } else {
1067  new_stmt =
1068  gimple_build_assign_with_ops(NOP_EXPR, oprnd0_tree,
1069  result_var, NULL);
1070  }
1071  gsi_insert_after(&gsi, new_stmt, GSI_NEW_STMT);
1072  print_gimple_stmt(stderr, new_stmt, 2, 0);
1073  }
1074  } // oprnd0 has fxfrmt attribute, must saturate
1075  } // result and oprnd0 are same size
1076  check_shift(result_frmt);
1077 
1078  if (AFFINE) {
1079  struct AA *new_aa = new_aa_list(result_frmt);
1080  delete_aa_list(&(result_frmt.aa));
1081  result_frmt.aa = new_aa;
1082  result_frmt.max = aa_max(result_frmt.aa);
1083  result_frmt.min = aa_min(result_frmt.aa);
1084  }
1085 
1086  //
1087  // Update and dump the statement, if modified
1088  //
1089  if (gimple_modified_p(stmt)) {
1090  update_stmt(stmt);
1091  print_gimple_stmt(stderr, stmt, 2, 0);
1092  }
1093  if (result_frmt.alias) {
1094  fprintf(stderr, " RSLT @");
1095  } else if (result_frmt.ptr_op) {
1096  fprintf(stderr, " RSLT *");
1097  } else {
1098  fprintf(stderr, " RSLT ");
1099  }
1100  fprintf(stderr, "(%2d/%2d/%2d/%2d)",
1101  result_frmt.S, result_frmt.I, result_frmt.F,
1102  result_frmt.E);
1103  if (result_frmt.sgnd) {
1104  fprintf(stderr, "s ");
1105  } else {
1106  fprintf(stderr, "u ");
1107  }
1108  if (AFFINE) {
1109  print_aa_list(result_frmt.aa);
1110  }
1111  if (INTERVAL)
1112  print_min_max(result_frmt);
1113  else
1114  fprintf(stderr, "\n");
1115 
1116  statements_updated++;
1117  } // result format is initialized
1118  else {
1119  undefined_result_format++;
1120  fprintf(stderr, " RESULT is uninitialized\n");
1121  } // result format is not initialized
1122  //
1123  // If there was a problem setting the LHS format, and the
1124  // LHS is a variable that is used iteratively, try again
1125  //
1126  if (set_var_format(oprnd0_tree, result_frmt) && oprnd_frmt[0].iter)
1127  {
1128  statements_updated++;
1129  undefined_result_format++;
1130  formats_changed++;
1131  } // LHS format changed
1132  delete_aa_list(&(result_frmt.aa));
1133  for (i = 0; i < num_operands; i++) {
1134  if (0xdead == oprnd_frmt[i].id)
1135  delete_aa_list(&(oprnd_frmt[i].aa));
1136  }
1137  } // an assign statement
1138  fprintf(stderr, "-------------------------------- %d %d\n",
1139  bbnumber, fxpass);
1140  if (gimple_code(stmt) == GIMPLE_RETURN) {
1141  tree return_val = gimple_return_retval(stmt);
1142  if (return_val != NULL_TREE) {
1143  if (SSA_NAME == TREE_CODE(return_val)) {
1144  if (lastpass) {
1145  tree return_var_tree = SSA_NAME_VAR(return_val);
1146  struct SIF *var_fmt;
1147  var_fmt = find_var_format(calc_hash_key(return_var_tree,
1148  fxpass, NOT_AN_ARRAY));
1149  if (format_initialized(*var_fmt)) {
1150  fprintf(stderr, "/// RETURNS %2d\n", var_fmt->S);
1151  fprintf(stderr, "/// RETURNI %2d\n", var_fmt->I);
1152  fprintf(stderr, "/// RETURNF %2d\n", var_fmt->F);
1153  fprintf(stderr, "/// RETURNE %2d\n", var_fmt->E);
1154  }
1155  }
1156  } else {
1157  fprintf(stderr, " Unexpected non-void return type");
1158  }
1159  } // non-void return value
